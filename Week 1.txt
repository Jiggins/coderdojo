 */
public class CoderDojo {
	
	@Instance("CoderDojo")
	public static CoderDojo instance;

	@SidedProxy(clientSide = "coderdojo.proxies.ClientProxy", serverSide = "coderdojo.proxies.CommonProxy")
	public static CommonProxy proxy;

	/**
     * 	FMLPreInitializationEvent : Run before anything else. Read your config, create blocks,
     * 	etc, and register them with the GameRegistry
     */
	@EventHandler
	public void preInit(FMLPreInitializationEvent event) {
		ConfigHandler.init(event.getSuggestedConfigurationFile());
		CreateBlocks.addBlocks();
	}

	/*
	 * Initialization
	 * Where Recipies and names are added to the game
	 */
	@EventHandler
	public void init(FMLInitializationEvent event) {
		CreateBlocks.addNames(); 
		CreateBlocks.registerRecipies();
	}
	
	/**
     * 	FMLPostInitializationEvent : Handle interaction with other mods, complete your setup based on this.
	 */
	@EventHandler
	public void postinit(FMLPostInitializationEvent event) {

	}
}

public class BlockConcrete extends Block {

	public BlockConcrete(int id) {
		super(id, Material.dragonEgg);
		
		setCreativeTab(CreativeTabs.tabBlock);
	}

}

public class ItemBlockConcrete extends ItemBlock {
	public ItemBlockConcrete(int id) {
		super(id);
	}
}

public class CreateBlocks {

	public static Block concrete;
	
	public static void addBlocks() {
		concrete = new BlockConcrete(1024);
		GameRegistry.registerBlock(concrete, ItemConcrete.class, BlockInfo.CONCRETE_KEY);
	}
	
	public static void addNames() {
		LanguageRegistry.addName(concrete, BlockInfo.CONCRETE_NAME);
	}

	public static void addRecipies() {
		ItemStack concreteStack = new ItemStack(concrete, 2);
		ItemStack sandStack = new ItemStack(Block.sand);
		ItemStack gravelStack = new ItemStack(Block.gravel);
		
		GameRegistry.addRecipe(concreteStack, 	"SG ",
												"GS ",
												"   ",
												'S', sandStack,
												'G', gravelStack);
		
		GameRegistry.addShapelessRecipe(concreteStack, sandStack, gravelStack);
		
		GameRegistry.addSmelting(Block.stone.blockID, concreteStack, 0.2F);
	}

}
